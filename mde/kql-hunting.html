Advanced Hunting

Use KQL in Microsoft Defender to hunt for threats across endpoints, emails, identities, and cloud apps. The same query language as Sentinel but with MDE-specific tables.

Hunting Queries

Process Hunting
// PowerShell with encoded commands
DeviceProcessEvents
| where Timestamp > ago(7d)
| where FileName =~ "powershell.exe" or FileName =~ "pwsh.exe"
| where ProcessCommandLine has_any ("-enc", "-encoded", "-e ", "-ec ")
| project Timestamp, DeviceName, AccountName, ProcessCommandLine, 
    InitiatingProcessFileName, InitiatingProcessCommandLine

// Suspicious process spawning from Office
DeviceProcessEvents
| where Timestamp > ago(24h)
| where InitiatingProcessFileName in~ ("winword.exe", "excel.exe", "powerpnt.exe", "outlook.exe")
| where FileName in~ ("cmd.exe", "powershell.exe", "wscript.exe", "cscript.exe", "mshta.exe")
| project Timestamp, DeviceName, AccountName, InitiatingProcessFileName, 
    FileName, ProcessCommandLine

// LOLBins usage
DeviceProcessEvents
| where Timestamp > ago(7d)
| where FileName in~ ("certutil.exe", "bitsadmin.exe", "mshta.exe", 
    "regsvr32.exe", "rundll32.exe", "msiexec.exe")
| where ProcessCommandLine has_any ("http", "ftp", "download", "transfer")
| project Timestamp, DeviceName, AccountName, FileName, ProcessCommandLine
Network & Persistence Hunting
// Connections to rare external IPs
DeviceNetworkEvents
| where Timestamp > ago(24h)
| where RemoteIPType == "Public"
| summarize ConnectionCount = count(), Devices = dcount(DeviceName) by RemoteIP
| where Devices == 1 and ConnectionCount < 5
| join kind=inner (
    DeviceNetworkEvents | where Timestamp > ago(24h)
) on RemoteIP
| project Timestamp, DeviceName, RemoteIP, RemotePort, InitiatingProcessFileName

// Scheduled task creation
DeviceRegistryEvents
| where Timestamp > ago(7d)
| where RegistryKey has "Schedule\\TaskCache\\Tasks"
| where ActionType == "RegistryValueSet"
| project Timestamp, DeviceName, RegistryKey, RegistryValueName, PreviousRegistryValueData

// Run key persistence
DeviceRegistryEvents
| where Timestamp > ago(7d)
| where RegistryKey has_any ("Run", "RunOnce")
| where ActionType == "RegistryValueSet"
| project Timestamp, DeviceName, AccountName, RegistryKey, RegistryValueName, RegistryValueData

Interview Questions

// Hunt for lateral movement indicators
// 1. Remote service creation
DeviceProcessEvents
| where Timestamp > ago(24h)
| where FileName == "services.exe"
| where ProcessCommandLine has "create" or ProcessCommandLine has "config"
| project Timestamp, DeviceName, ProcessCommandLine

// 2. PsExec-like activity
DeviceNetworkEvents
| where Timestamp > ago(24h)
| where RemotePort == 445 or RemotePort == 135
| summarize TargetCount = dcount(RemoteIP) by DeviceName
| where TargetCount > 5

// 3. WMI remote execution
DeviceProcessEvents
| where Timestamp > ago(24h)
| where InitiatingProcessFileName == "wmiprvse.exe"
| where FileName in~ ("cmd.exe", "powershell.exe")
