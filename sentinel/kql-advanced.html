KQL Advanced Techniques

Master advanced KQL patterns for threat hunting, complex analytics, and optimized queries in Microsoft Sentinel.

Advanced Join Patterns

Inner Join with Multiple Keys

// Join sign-in logs with audit logs SigninLogs | where TimeGenerated > ago(24h) | where ResultType == 0 // Successful | join kind=inner ( AuditLogs | where TimeGenerated > ago(24h) | where OperationName has "Add member to group" | extend TargetUser = tostring(TargetResources[0].userPrincipalName) ) on $left.UserPrincipalName == $right.TargetUser | project TimeGenerated, UserPrincipalName, IPAddress, OperationName

Anti-Join (Find Non-Matching)

// Find IPs that authenticated but never appeared before let known_ips = SigninLogs | where TimeGenerated between (ago(30d) .. ago(1d)) | distinct IPAddress; SigninLogs | where TimeGenerated > ago(1d) | where ResultType == 0 | where IPAddress !in (known_ips) | summarize FirstSeen=min(TimeGenerated), Count=count() by UserPrincipalName, IPAddress

Left Outer Join with Enrichment

// Enrich security events with threat intelligence SecurityEvent | where TimeGenerated > ago(1h) | where EventID == 4625 | extend IPAddress = IpAddress | join kind=leftouter ( ThreatIntelligenceIndicator | where TimeGenerated > ago(30d) | where isnotempty(NetworkIP) | project TI_IP=NetworkIP, ThreatType, ConfidenceScore ) on $left.IPAddress == $right.TI_IP | where isnotempty(ThreatType) | project TimeGenerated, Account, IPAddress, ThreatType, ConfidenceScore

Subqueries and Let Statements

Dynamic Thresholds with let

// Calculate dynamic baseline and detect anomalies let baseline = SecurityEvent | where TimeGenerated between (ago(7d) .. ago(1d)) | where EventID == 4625 | summarize avg_failures = avg(todouble(count())) by bin(TimeGenerated, 1h); let threshold = toscalar(baseline | summarize avg(avg_failures) * 3); SecurityEvent | where TimeGenerated > ago(1h) | where EventID == 4625 | summarize failures = count() by bin(TimeGenerated, 5m), Computer | where failures > threshold | project TimeGenerated, Computer, failures, Threshold=threshold

Reusable Functions with let

// Define reusable function let GetRiskScore = (ip:string) { ThreatIntelligenceIndicator | where NetworkIP == ip | summarize MaxScore = max(ConfidenceScore) | project coalesce(MaxScore, 0) }; SigninLogs | where TimeGenerated > ago(1h) | extend RiskScore = GetRiskScore(IPAddress) | where RiskScore > 70 | project TimeGenerated, UserPrincipalName, IPAddress, RiskScore

Materialized Views Pattern

// Materialize intermediate results for reuse let suspicious_users = materialize( SigninLogs | where TimeGenerated > ago(24h) | where ResultType != 0 | summarize FailedCount = count() by UserPrincipalName | where FailedCount > 10 ); suspicious_users | join kind=inner ( SigninLogs | where TimeGenerated > ago(24h) | where ResultType == 0 ) on UserPrincipalName | project TimeGenerated, UserPrincipalName, FailedCount, SuccessfulIP=IPAddress

Time Series Analysis

make-series for Trend Analysis

// Create time series with gaps filled SigninLogs | where TimeGenerated > ago(7d) | make-series LoginCount=count() default=0 on TimeGenerated step 1h | extend (anomalies, score, baseline) = series_decompose_anomalies(LoginCount, 1.5) | mv-expand TimeGenerated, LoginCount, anomalies, score, baseline | where anomalies == 1 | project TimeGenerated=todatetime(TimeGenerated), LoginCount=toint(LoginCount), AnomalyScore=todouble(score)

Series Functions

// Detect spikes using series statistics SecurityEvent | where TimeGenerated > ago(30d) | where EventID in (4624, 4625) | make-series EventCount=count() default=0 on TimeGenerated step 1h by Computer | extend (flag, score) = series_outliers(EventCount) | mv-expand TimeGenerated, EventCount, flag, score | where flag == 1 and score > 2 | project TimeGenerated=todatetime(TimeGenerated), Computer, EventCount=toint(EventCount), OutlierScore=round(todouble(score), 2) // Moving average | make-series Count=count() on TimeGenerated step 1h | extend MovingAvg = series_fir(Count, repeat(1, 24), true, true) // 24-hour moving avg

Advanced Parsing

Complex Parse Patterns

// Parse nested JSON from string field AzureDiagnostics | extend Properties = parse_json(properties_s) | extend CallerIP = tostring(Properties.callerIpAddress) | extend OperationType = tostring(Properties.operationType) | extend ResourceName = tostring(Properties.resourceName) // Parse with multiple patterns (try each until match) Syslog | parse SyslogMessage with * "user=" User " " * | parse SyslogMessage with * "src=" SrcIP ":" SrcPort " " * | parse SyslogMessage with * "action=" Action * | where isnotempty(User) // Extract using regex SecurityEvent | extend CommandLine = extract(@"CommandLine:\s*(.+?)(?:\s+|$)", 1, CommandLine) | extend ParentProcess = extract(@"ParentProcessName:\s*(\S+)", 1, Message)

Dynamic JSON Parsing

// Parse dynamic JSON and extract nested values let data = datatable(json_str:string) [ '{"user":{"name":"john","id":123},"action":"login"}', '{"user":{"name":"jane","id":456},"action":"logout"}' ]; data | extend parsed = parse_json(json_str) | extend UserName = tostring(parsed.user.name) | extend UserId = toint(parsed.user.id) | extend Action = tostring(parsed.action) | project UserName, UserId, Action // Handle arrays in JSON AuditLogs | extend TargetResources = parse_json(TargetResources) | mv-expand TargetResource = TargetResources | extend TargetName = tostring(TargetResource.displayName) | extend TargetType = tostring(TargetResource.type)

Threat Hunting Patterns

Beacon Detection (C2 Communication)

// Detect regular interval connections (beaconing) CommonSecurityLog | where TimeGenerated > ago(24h) | where DeviceAction == "allow" | summarize ConnectionTimes = make_list(TimeGenerated), Count = count() by SourceIP, DestinationIP, DestinationPort | where Count > 50 | extend Intervals = series_stats_dynamic( array_sort_asc( series_subtract( array_slice(ConnectionTimes, 1, -1), array_slice(ConnectionTimes, 0, -2) ) ) ) | extend AvgInterval = Intervals.avg, StdDev = Intervals.stdev | where StdDev < AvgInterval * 0.1 // Low variance = regular beaconing | project SourceIP, DestinationIP, DestinationPort, Count, AvgIntervalSec = AvgInterval/10000000, StdDevSec = StdDev/10000000

Lateral Movement Detection

// Detect user authenticating to many machines SecurityEvent | where TimeGenerated > ago(1h) | where EventID == 4624 | where LogonType in (3, 10) // Network, RemoteInteractive | summarize TargetMachines = make_set(Computer), MachineCount = dcount(Computer) by Account, IpAddress | where MachineCount > 5 | project Account, IpAddress, MachineCount, TargetMachines // Chain of authentications SecurityEvent | where TimeGenerated > ago(24h) | where EventID == 4624 | where LogonType == 3 | project TimeGenerated, SourceComputer=WorkstationName, TargetComputer=Computer, Account | join kind=inner ( SecurityEvent | where TimeGenerated > ago(24h) | where EventID == 4624 | where LogonType == 3 | project TimeGenerated2=TimeGenerated, SourceComputer2=WorkstationName, TargetComputer2=Computer, Account ) on $left.TargetComputer == $right.SourceComputer2, Account | where TimeGenerated2 > TimeGenerated and TimeGenerated2 < TimeGenerated + 5m | project TimeGenerated, Account, Hop1=SourceComputer, Hop2=TargetComputer, Hop3=TargetComputer2

Data Exfiltration Detection

// Large outbound data transfers CommonSecurityLog | where TimeGenerated > ago(24h) | where DeviceAction == "allow" | where DestinationPort in (80, 443) | summarize TotalBytes = sum(SentBytes), Connections = count() by SourceIP, DestinationIP, bin(TimeGenerated, 1h) | where TotalBytes > 100000000 // 100 MB | join kind=leftouter ( ThreatIntelligenceIndicator | where isnotempty(NetworkIP) | project NetworkIP, ThreatType ) on $left.DestinationIP == $right.NetworkIP | project TimeGenerated, SourceIP, DestinationIP, TotalBytesMB=TotalBytes/1048576, Connections, ThreatType

Query Optimization

Optimization Techniques

// BAD: Inefficient query AzureActivity | extend parsed = parse_json(Properties) | where parsed.statusCode == "OK" | project TimeGenerated, Caller // GOOD: Filter first, then parse AzureActivity | where TimeGenerated > ago(1h) | where ActivityStatus == "Succeeded" | extend parsed = parse_json(Properties) | project TimeGenerated, Caller // Use has instead of contains for better performance // BAD | where CommandLine contains "powershell" // GOOD | where CommandLine has "powershell" // Avoid wildcards at start // BAD | where ProcessName matches regex ".*powershell.*" // GOOD | where ProcessName has "powershell" // Limit join data let small_table = SigninLogs | where TimeGenerated > ago(1h) | where ResultType != 0 | summarize by UserPrincipalName; SigninLogs | where TimeGenerated > ago(1h) | where UserPrincipalName in (small_table)

Interview Questions

Approach:

  1. Aggregate connections by source, dest, port with timestamps
  2. Calculate time intervals between connections
  3. Use series_stats_dynamic to get average and standard deviation
  4. Low standard deviation relative to average = regular interval = beaconing

Key insight: C2 beacons have predictable timing. Normal traffic is random.

  • Filter early: Put where clauses before expensive operations
  • Use has over contains: has uses index, contains doesn't
  • Avoid regex when possible: Use startswith, endswith, has
  • Limit join sizes: Aggregate or filter before joining
  • Use materialize(): When referencing same subquery multiple times
  • Project early: Remove unneeded columns

Additional Interview Questions

Systematic troubleshooting approach:

  1. Identify the symptom: What exactly is not working as expected?
  2. Check connectivity: Verify network paths and authentication
  3. Review logs: Check internal/audit logs for errors
  4. Validate configuration: Compare against documented standards
  5. Test incrementally: Isolate the problem component
  6. Document findings: Record what worked and what didn't

Key performance indicators:

  • Availability: Uptime percentage, service availability
  • Performance: Query response time, processing latency
  • Volume: Events per second, data ingestion rate
  • Quality: Parse success rate, enrichment coverage
  • Efficiency: Resource utilization, cost per GB

High availability strategies:

  • Redundancy: Multiple instances of critical components
  • Load balancing: Distribute traffic across instances
  • Failover: Automatic switchover on failure
  • Monitoring: Health checks and alerting
  • DR Planning: Documented recovery procedures
