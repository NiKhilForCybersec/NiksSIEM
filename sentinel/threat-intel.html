TI Matching Query

Match Indicators Against Logs
// Match IPs against threat intel
let TI_IPs = ThreatIntelligenceIndicator
| where Active == true
| where ExpirationDateTime > now()
| where NetworkIP != ""
| summarize TIData = make_set(AdditionalInformation) by NetworkIP;
SigninLogs
| where TimeGenerated > ago(1h)
| join kind=inner TI_IPs on $left.IPAddress == $right.NetworkIP
| project TimeGenerated, UserPrincipalName, IPAddress, TIData

// Match domains against DNS logs
let TI_Domains = ThreatIntelligenceIndicator
| where DomainName != ""
| distinct DomainName;
DnsEvents
| where TimeGenerated > ago(1h)
| where Name in (TI_Domains)
| project TimeGenerated, Computer, Name, IPAddresses
